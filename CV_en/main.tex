\def\lmarginbar{-4.2pt} % left margin of vertical line from the marker

% 2. Custom Timeline Environment (The connecting line is now here)
\newtcolorbox{timelinebox}[1][]{
    blanker,
    breakable,
    left=1.5em,
    overlay={
        % If NOT the last item, draw the connecting line downwards
        \ifthenelse{\equal{#1}{last}}{
            \draw[theme_color, line width=1pt] 
                ([xshift=\lmarginbar, yshift=-5pt]frame.north west) -- ([xshift=\lmarginbar]frame.south west);
        }{
            % Default length (10pt)
            \draw[theme_color, line width=1pt] 
                ([xshift=\lmarginbar, yshift=-5pt]frame.north west) -- ([xshift=\lmarginbar, yshift=-25pt]frame.south west);
        }
    }
}

% 3. The New cvevent Command
\newcommand{\cvevent}[5]{%
    \noindent
    % --- COLUMN 1: Date and Marker ---
    \begin{minipage}[t]{\leftcolwidth}
        \raggedleft
        {\small\color{gray}#3}\hspace{1em} % The Date
        % The Marker
        \begin{tikzpicture}[baseline=-3pt]
            \node(circle) [circle, fill=theme_color, inner sep=3pt] {};
        \end{tikzpicture}
    \end{minipage}%
    % --- COLUMN 2: Content ---
    \begin{minipage}[t]{\dimexpr\textwidth-\leftcolwidth\relax}
        \begin{timelinebox}[#4]
            \textbf{\color{header_color}#1}\par
            {\color{gray}\small \uppercase{#2}}\par
            \if\relax\detokenize{#5}\relax
            \else
                \begin{itemize}[leftmargin=1.5em, label={\color{header_color}\textbullet}, itemsep=0.2em, parsep=0pt]
                #5
                \end{itemize}
            \fi
        \end{timelinebox}
    \end{minipage}
    \vspace{2ex}
}
